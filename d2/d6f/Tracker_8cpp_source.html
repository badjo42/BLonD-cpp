57 inline void RingAndRfSection::kick(const ftype *__restrict beam_dt,
58  ftype *__restrict beam_dE,
59  const int n_rf,
60  const ftype *__restrict voltage,
61  const ftype *__restrict omega_RF,
62  const ftype *__restrict phi_RF,
63  const int n_macroparticles,
64  const ftype acc_kick)
65 {
66  // KICK
67  //#pragma omp parallel for collapse(2)
68  for (int j = 0; j < n_rf; ++j) {
69  #pragma omp parallel for
70  for (int i = 0; i < n_macroparticles; ++i) {
71  const ftype a = omega_RF[j] * beam_dt[i] + phi_RF[j];
72  beam_dE[i] += voltage[j] * mymath::fast_sin(a);
73  }
74  }
75 
76 // SYNCHRONOUS ENERGY CHANGE
77  #pragma omp parallel for
78  for (int i = 0; i < n_macroparticles; ++i)
79  beam_dE[i] += acc_kick;
80 }

83 inline void RingAndRfSection::drift(ftype *__restrict beam_dt,
84  const ftype *__restrict beam_dE,
85  const solver_type solver,
86  const ftype T0,
87  const ftype length_ratio,
88  const uint alpha_order,
89  const ftype eta_zero,
90  const ftype eta_one,
91  const ftype eta_two,
92  const ftype beta,
93  const ftype energy,
94  const int n_macroparticles)
95 {
96 
97  const ftype T = T0 * length_ratio;
98 
99  if (solver == simple) {
100  const ftype T_x_coeff = T * eta_zero / (beta * beta * energy);
101  #pragma omp parallel for
102  for (int i = 0; i < n_macroparticles; i++)
103  beam_dt[i] += T_x_coeff * beam_dE[i];
104  } else {
105  const ftype coeff = 1. / (beta * beta * energy);
106  const ftype eta0 = eta_zero * coeff;
107  const ftype eta1 = eta_one * coeff * coeff;
108  const ftype eta2 = eta_two * coeff * coeff * coeff;
109 
110  if (alpha_order == 1)
111  #pragma omp parallel for
112  for (int i = 0; i < n_macroparticles; i++)
113  beam_dt[i] += T * (1. / (1. - eta0 * beam_dE[i]) - 1.);
114  else if (alpha_order == 2)
115  #pragma omp parallel for
116  for (int i = 0; i < n_macroparticles; i++)
117  beam_dt[i] += T * (1. / (1. - eta0 * beam_dE[i] -
118  eta1 * beam_dE[i] * beam_dE[i]) -
119  1.);
120  else
121  #pragma omp parallel for
122  for (int i = 0; i < n_macroparticles; i++)
123  beam_dt[i] +=
124  T * (1. / (1. - eta0 * beam_dE[i] -
125  eta1 * beam_dE[i] * beam_dE[i] -
126  eta2 * beam_dE[i] * beam_dE[i] * beam_dE[i]) -
127  1.);
128  }
129 }

133 void RingAndRfSection::track()
134 {
135  auto GP = Context::GP;
136  auto RfP = fRfP;
137  auto Beam = Context::Beam;
138 
139  if (!RfP->phi_noise.empty()) {
140  if (noiseFB != NULL) {
141  for (uint i = 0; i < RfP->phi_RF.size(); ++i)
142  RfP->phi_RF[i][RfP->counter] +=
143  noiseFB->fX * RfP->phi_noise[i][RfP->counter];
144  } else {
145  for (uint i = 0; i < RfP->phi_RF.size(); ++i)
146  RfP->phi_RF[i][RfP->counter] += RfP->phi_noise[i][RfP->counter];
147  }
148  }
149 
150  // Determine phase loop correction on RF phase and frequency
151  if (PL != NULL && RfP->counter >= PL->delay)
152  PL->track();
153 
154  if (periodicity) {
155  // Change reference of all the particles on the right of the current
156  // frame; these particles skip one kick and drift
157  set_periodicity();
158  const ftype tRev = GP->t_rev[RfP->counter + 1];
159 
160  if (!indices_right_outside.empty()) {
161  f_vector_t insiders_dt, insiders_dE;
162  insiders_dt.reserve(indices_inside_frame.size());
163  insiders_dE.reserve(indices_inside_frame.size());
164  for (const auto &i : indices_inside_frame) {
165  insiders_dt.push_back(Beam->dt[i]);
166  insiders_dE.push_back(Beam->dE[i]);
167  }
168 
169  for (const auto &i : indices_right_outside)
170  Beam->dt[i] -= tRev;
171 
172  // Synchronize the bunch with the particles that are on the right of
173  // the current frame applying kick and drift to the bunch; after that
174  // all the particle are in the new updated frame
175  kick(insiders_dt, insiders_dE, RfP->counter);
176  drift(insiders_dt, insiders_dE, RfP->counter + 1);
177  int k = 0;
178  for (const auto &i : indices_inside_frame) {
179  Beam->dt[i] = insiders_dt[k];
180  Beam->dE[i] = insiders_dE[k];
181  k++;
182  }
183 
184  } else {
185  kick(Beam->dt, Beam->dE, RfP->counter);
186  drift(Beam->dt, Beam->dE, RfP->counter + 1);
187  // find left outside particles and kick, drift them one more time
188  // indices_left_outside.clear();
189  //#pragma omp parallel for reduction(+:a)
190  for (uint i = 0; i < Beam->n_macroparticles; ++i) {
191  if (Beam->dt[i] < 0)
192  indices_left_outside.push_back(i);
193  }
194  }
195 
196  if (!indices_left_outside.empty()) {
197  f_vector_t left_dt, left_dE;
198  left_dt.reserve(indices_left_outside.size());
199  left_dE.reserve(indices_left_outside.size());
200  for (const auto &i : indices_left_outside) {
201  left_dt.push_back(Beam->dt[i] + tRev);
202  left_dE.push_back(Beam->dE[i]);
203  }
204 
205  kick(left_dt, left_dE, RfP->counter);
206  drift(left_dt, left_dE, RfP->counter + 1);
207  int k = 0;
208  for (const auto &i : indices_left_outside) {
209  Beam->dt[i] = left_dt[k];
210  Beam->dE[i] = left_dE[k];
211  k++;
212  }
213  }
214 
215  // std::cout << "right: " << indices_right_outside.size() << '\n';
216  // std::cout << "inside: " << indices_inside_frame.size() << '\n';
217  // std::cout << "left: " << indices_left_outside.size() << '\n';
218 
219  } else {
220  if (rf_kick_interp) {
221  // TODO test this part
222  rf_voltage_calculation(RfP->counter, slices);
223  fTotalVoltage = fRfVoltage;
224 
225  if (totalInducedVoltage != NULL) {
226  assert(fRfVoltage.size() ==
227  totalInducedVoltage->fInducedVoltage.size());
228  for (uint i = 0; i < fRfVoltage.size(); i++)
229  fTotalVoltage[i] += totalInducedVoltage->fInducedVoltage[i];
230  }
231 
232  std::transform(fRfVoltage.begin(), fRfVoltage.end(),
233  fRfVoltage.begin(),
234  std::bind1st(std::multiplies<ftype>(), GP->charge));
235 
236  linear_interp_kick(Beam->dt.data(), Beam->dE.data(),
237  fRfVoltage.data(), slices->bin_centers.data(),
238  slices->n_slices, Beam->n_macroparticles);
239  } else {
240  kick(Beam->dt, Beam->dE, RfP->counter);
241  }
242  drift(Beam->dt, Beam->dE, RfP->counter + 1);
243  }
244 
245  if (dE_max > 0) horizontal_cut();
246 
247  RfP->counter++;
248 }

269 void RingAndRfSection::rf_voltage_calculation(uint turn, Slices *slices)

271  // Calculating the RF voltage seen by the beam at a given turn,
272  // needs a Slices object.
273  auto RfP = fRfP;
274 
275  auto vol = new ftype[RfP->n_rf];

278

288  ftype sum = 0.0;

290  sum += vol[i] * std::sin(omeg[i] * slices->bin_centers[j] + phi[i]);
291  }
292  fRfVoltage[j] = sum;
293  }
294 
295 
296  delete[] vol;
297  delete[] omeg;
298  delete[] phi;

380 void FullRingAndRf::potential_well_generation(const uint turn,
381  const uint n_points,
382  const ftype option,
383  const ftype dt_margin_percent)
384 {
385  auto GP = Context::GP;
386  f_vector_t voltages;
387  f_vector_t omega_rf;
388  f_vector_t phi_offsets;
389 
390  for (const auto &ring : fRingList) {
391  for (uint i = 0; i < ring->fRfP->n_rf; ++i) {
392  voltages.push_back(ring->fRfP->voltage[i][turn]);
393  omega_rf.push_back(ring->fRfP->omega_RF[i][turn]);
394  phi_offsets.push_back(ring->fRfP->phi_RF[i][turn]);
395  }
396  }
397 
398  ftype main_omega_rf;
399  if (option == 0) {
400  auto k = mymath::min(omega_rf.data(), omega_rf.size());
401  main_omega_rf = omega_rf[k];
402  } else if (option == 1) {
403  auto k = mymath::max(voltages.data(), voltages.size());
404  auto maxV = voltages[k];
405 
406  f_vector_t temp;
407 
408  for (uint i = 0; i < voltages.size(); ++i)
409  if (voltages[i] == maxV)
410  temp.push_back(omega_rf[i]);
411 
412  k = mymath::min(temp.data(), temp.size());
413  main_omega_rf = temp[k];
414  } else {
415  f_vector_t temp;
416  std::copy_if(omega_rf.begin(), omega_rf.end(), back_inserter(temp),
417  [option](const ftype x) { return x == option; });
418  if (temp.empty()) {
419  std::cerr << "[ERROR] The desired harmonic to compute"
420  << "the potential well does not"
421  << "match the RF parameters...\n";
422  exit(-1);
423  }
424  main_omega_rf = option;//temp[k];
425  }
426 
427  auto time_array_margin =
428  dt_margin_percent * 2 * constant::pi / main_omega_rf;
429 
430  auto slippage_factor = fRingList[0]->fRfP->eta_0(turn);
431 
432  auto first_dt = -time_array_margin / 2;
433  auto last_dt = 2 * constant::pi / main_omega_rf + time_array_margin / 2;
434 
435  f_vector_t time_array(n_points);
436  mymath::linspace(time_array.data(), first_dt, last_dt, n_points);
437 
438  fTotalVoltage.resize(time_array.size());
439 
440  for (int i = 0; i < (int)time_array.size(); ++i) {
441  ftype sum = 0.0;
442  for (int j = 0; j < (int)voltages.size(); ++j) {
443  sum += voltages[j] * mymath::fast_sin(omega_rf[j] * time_array[i] +
444  phi_offsets[j]);
445  }
446  fTotalVoltage[i] = sum;
447  }
448 
449  const auto eom_factor_potential =
450  mymath::sign(slippage_factor) * GP->charge / GP->t_rev[turn];
451 
452  auto tempVec = fTotalVoltage;
453  for (int i = 0; i < (int)fTotalVoltage.size(); ++i)
454  tempVec[i] =
455  eom_factor_potential *
456  (tempVec[i] -
457  (-fRingList[0]->acceleration_kick[turn]) / std::abs(GP->charge));
458 
459  auto tempArr = mymath::cum_trapezoid(
460  tempVec.data(), time_array[1] - time_array[0], n_points);
461 
462  std::transform(tempArr.begin(), tempArr.end(), tempArr.begin(),
463  [](ftype x) { return -x; });
464 
465  auto min_i = mymath::min(tempArr.data(), tempArr.size());
466  auto min = tempArr[min_i] > 0 ? 0 : tempArr[min_i];
467 
468  fPotentialWell.resize(tempArr.size() + 1);
469  fPotentialWell[0] = - min;
470 
471  for (uint i = 0; i < tempArr.size(); ++i)
472  fPotentialWell[i + 1] = tempArr[i] - min;
473 
474  fPotentialWellCoordinates = time_array;
475 }
