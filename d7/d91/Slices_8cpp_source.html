40 void Slices::set_cuts()
41 {
42  auto Beam = Context::Beam;
43  /*
44  *Method to set the self.cut_left and self.cut_right properties. This is
45  done as a pre-processing if the mode is set to 'const_space', for
46  'const_charge' this is calculated each turn.*
47  *The frame is defined by :math:`n\sigma_{RMS}` or manually by the user.
48  If not, a default frame consisting of taking the whole bunch +5% of the
49  maximum distance between two particles in the bunch will be taken
50  in each side of the frame.*
51  */
52 
53  if (cut_left == 0 && cut_right == 0) {
54  if (n_sigma == 0) {
55  sort_particles();
56  cut_left =
57  Beam->dt.front() - 0.05 * (Beam->dt.back() - Beam->dt.front());
58  cut_right =
59  Beam->dt.back() + 0.05 * (Beam->dt.back() - Beam->dt.front());
60  } else {
61  ftype mean_coords = mymath::mean(Beam->dt.data(), Beam->dt.size());
62  ftype sigma_coords = mymath::standard_deviation(
63  Beam->dt.data(), Beam->dt.size(), mean_coords);
64  cut_left = mean_coords - n_sigma * sigma_coords / 2;
65  cut_right = mean_coords + n_sigma * sigma_coords / 2;
66  }
67  } else {
68  cut_left = convert_coordinates(cut_left, cuts_unit);
69  cut_right = convert_coordinates(cut_right, cuts_unit);
70  }
71 
72  mymath::linspace(edges.data(), cut_left, cut_right, n_slices + 1);
73  for (uint i = 0; i < bin_centers.size(); ++i)
74  bin_centers[i] = (edges[i + 1] + edges[i]) / 2;
75 }

77 void Slices::sort_particles()
78 {
79  /*
80  *Sort the particles with respect to their position.*
81  */
82 
83  auto Beam = Context::Beam;
84  struct API particle {
85  ftype dE;
86  ftype dt;
87  int id;
88  bool operator<(const particle &other) const
89  {
90  return dt < other.dt;
91  }
92  };
93 
94  std::vector<particle> particles(Beam->dE.size());
95  for (uint i = 0; i < particles.size(); i++)
96  particles[i] = {Beam->dE[i], Beam->dt[i], Beam->id[i]};
97 
98  std::sort(particles.begin(), particles.end());
99 
100  for (uint i = 0; i < particles.size(); i++) {
101  Beam->dE[i] = particles[i].dE;
102  Beam->dt[i] = particles[i].dt;
103  Beam->id[i] = particles[i].id;
104  }
105 }

147 inline void Slices::histogram(const ftype *__restrict input,
148  int *__restrict output, const ftype cut_left,
149  const ftype cut_right, const uint n_slices,
150  const uint n_macroparticles)
151 {
152 
153  const ftype inv_bin_width = n_slices / (cut_right - cut_left);
154  // const ftype range = cut_right - cut_left;
155  // histogram is faster with ints
156  typedef int hist_t;
157  hist_t *h;
158  #pragma omp parallel
159  {
160  const auto threads = omp_get_num_threads();
161  const auto id = omp_get_thread_num();
162  auto tile =
163  static_cast<int>((n_macroparticles + threads - 1) / threads);
164  auto start = id * tile;
165  auto end = std::min(start + tile, (int)n_macroparticles);
166  const auto row = id * n_slices;
167 
168  #pragma omp single
169  h = (hist_t *)calloc(threads * n_slices, sizeof(hist_t));
170 
171  const auto h_row = &h[row];
172 
173  for (int i = start; i < end; ++i) {
174  const ftype a = input[i];
175  // const ftype a = (input[i] - cut_left);
176  if (a < cut_left || a > cut_right)
177  continue;
178  const uint ffbin =
179  static_cast<uint>((a - cut_left) * inv_bin_width);
180 
181  // if (a < 0)
182  // continue;
183 
184  // const uint ffbin = static_cast<uint>(a * inv_bin_width);
185 
186  // if (ffbin >= n_slices)
187  // continue;
188 
189  h_row[ffbin]++;
190  }
191  #pragma omp barrier
192 
193  tile = (n_slices + threads - 1) / threads;
194  start = id * tile;
195  end = std::min(start + tile, (int)n_slices);
196 
197  for (int i = start; i < end; i++)
198  output[i] = 0;
199  // memset(&output[start], 0, (end-start) * sizeof(ftype));
200 
201  for (int i = 0; i < threads; ++i) {
202  const int r = i * n_slices;
203  for (int j = start; j < end; ++j) {
204  output[j] += h[r + j];
205  }
206  }
207  }
208  if (h)
209  free(h);
210 }

212 void Slices::track_cuts()
213 {
214  /*
215  *Track the slice frame (limits and slice position) as the mean of the
216  bunch moves.
217  Requires Beam statistics!
218  Method to be refined!*
219  */
220  auto Beam = Context::Beam;
221 
222  ftype delta = Beam->mean_dt - 0.5 * (cut_left + cut_right);
223  cut_left += delta;
224  cut_right += delta;
225  for (uint i = 0; i < n_slices + 1; ++i) {
226  edges[i] += delta;
227  }
228  for (uint i = 0; i < n_slices; ++i) {
229  bin_centers[i] += delta;
230  }

233 inline void Slices::smooth_histogram(const ftype *__restrict input,
234  int *__restrict output,
235  const ftype cut_left,
236  const ftype cut_right, const uint n_slices,
237  const uint n_macroparticles)
238 {
239 
240  uint i;
241  ftype a;
242  ftype fbin;
243  ftype ratioffbin;
244  ftype ratiofffbin;
245  ftype distToCenter;
246  uint ffbin = 0;
247  uint fffbin = 0;
248  const ftype inv_bin_width = n_slices / (cut_right - cut_left);
249  const ftype bin_width = (cut_right - cut_left) / n_slices;
250 
251  for (i = 0; i < n_slices; i++) {
252  output[i] = 0;
253  }
254 
255  for (i = 0; i < n_macroparticles; i++) {
256  a = input[i];
257  if ((a < (cut_left + bin_width * 0.5)) ||
258  (a > (cut_right - bin_width * 0.5)))
259  continue;
260  fbin = (a - cut_left) * inv_bin_width;
261  ffbin = (uint)(fbin);
262  distToCenter = fbin - (ftype)(ffbin);
263  if (distToCenter > 0.5)
264  fffbin = (uint)(fbin + 1.0);
265  ratioffbin = 1.5 - distToCenter;
266  ratiofffbin = 1 - ratioffbin;
267  if (distToCenter < 0.5)
268  fffbin = (uint)(fbin - 1.0);
269  ratioffbin = 0.5 - distToCenter;
270  ratiofffbin = 1 - ratioffbin;
271  output[ffbin] = output[ffbin] + ratioffbin;
272  output[fffbin] = output[fffbin] + ratiofffbin;
273  }

287 void Slices::rms()
288 {
289 
290  /*
291  * Computation of the RMS bunch length and position from the line density
292  (bunch length = 4sigma).*
293  */
294  auto Beam = Context::Beam;
295 
296  f_vector_t lineDenNormalized(n_slices); // = new ftype[n_slices];
297  f_vector_t array(n_slices); // = new ftype[n_slices];
298 
299  ftype timeResolution = bin_centers[1] - bin_centers[0];
300  ftype trap = mymath::trapezoid(n_macroparticles.data(), timeResolution,
301  Beam->n_macroparticles);
302 
303  for (uint i = 0; i < n_slices; ++i)
304  lineDenNormalized[i] = n_macroparticles[i] / trap;
305 
306  for (uint i = 0; i < n_slices; ++i)
307  array[i] = bin_centers[i] * lineDenNormalized[i];
308 
309  bp_rms = mymath::trapezoid(array.data(), timeResolution, n_slices);
310 
311  for (uint i = 0; i < n_slices; ++i)
312  array[i] = (bin_centers[i] - bp_rms) * (bin_centers[i] - bp_rms) *
313  lineDenNormalized[i];
314 
315  ftype temp = mymath::trapezoid(array.data(), timeResolution, n_slices);
316  bl_rms = 4 * std::sqrt(temp);
317 }

319 void Slices::fwhm(const ftype shift)
320 {
321 
322  /*
323  * Computation of the bunch length and position from the FWHM
324  assuming Gaussian line density.*
325  */
326  uint max_i = mymath::max(n_macroparticles.data(), n_slices, 1);
327  ftype half_max = shift + 0.5 * (n_macroparticles[max_i] - shift);
328  ftype timeResolution = bin_centers[1] - bin_centers[0];
329 
330  // First aproximation for the half maximum values
331 
332  int i = 0;
333  while (n_macroparticles[i] < half_max && i < (int)n_slices)
334  i++;
335  int taux1 = i;
336  i = n_slices - 1;
337  while (i >= 0 && n_macroparticles[i] < half_max)
338  i--;
339  int taux2 = i;
340 
341  // dprintf("taux1, taux2 = %d, %d\n", taux1, taux2);
342  ftype t1, t2;
343 
344  // maybe we could specify what kind of exceptions may occur here
345  // numerical (divide by zero) or index out of bounds
346 
347  // TODO something weird is happening here
348  // Python throws an exception only if you access an element after the end of
349  // the array
350  // but not if you access element before the start of an array
351  // (in that case it takes the last element of the array)
352  // Cpp does not throw an exception on eiter occassion
353  // The right condition is the following in comments
354  if (taux1 > 0 && taux2 < (int)n_slices - 1) {
355  // if (taux2 < n_slices - 1) {
356  try {
357  t1 = bin_centers[taux1] -
358  (n_macroparticles[taux1] - half_max) /
359  (n_macroparticles[taux1] - n_macroparticles[taux1 - 1]) *
360  timeResolution;
361 
362  t2 = bin_centers[taux2] +
363  (n_macroparticles[taux2] - half_max) /
364  (n_macroparticles[taux2] - n_macroparticles[taux2 + 1]) *
365  timeResolution;
366  bl_fwhm = 4 * (t2 - t1) / cfwhm;
367  bp_fwhm = (t1 + t2) / 2;
368  } catch (...) {
369  bl_fwhm = nan("");
370  bp_fwhm = nan("");
371  }
372  } else {

420 void Slices::beam_profile_derivative(f_vector_t &x,
421  f_vector_t &derivative,
422  std::string mode)
423 {
424  /*
425  The input is one of the two available methods for differentiating
426  a function. The two outputs are the coordinate step and the discrete
427  derivative of the Beam profile respectively.
428  */
429 
430  x = bin_centers;
431  const auto dist_centers = x[1] - x[0];
432  if (mode == "filter1d") {
433 
434  f_vector_t temp(n_macroparticles.begin(), n_macroparticles.end());
435  derivative = gaussian_filter1d(temp, 1, 1, "wrap");
436  for (auto &d : derivative) d /= dist_centers;
437 
438  } else if (mode == "gradient") {
439 
440  f_vector_t temp(n_macroparticles.begin(), n_macroparticles.end());
441  derivative = gradient(temp, dist_centers);
442 
443  } else if (mode == "diff") {
444 
445  derivative.resize(n_macroparticles.size() - 1);
446  for (uint i = 1; i < n_macroparticles.size(); i++)
447  derivative[i - 1] = (n_macroparticles[i] - n_macroparticles[i - 1])
448  / dist_centers;
449  f_vector_t diffCenters(x.begin(), x.end() - 1);
450  for (auto &d : diffCenters) d += dist_centers / 2;
451  f_vector_t res;
452  mymath::lin_interp(x, diffCenters, derivative,
453  res, derivative.front(), derivative.back());
454  derivative = res;
455  } else {
456  std::cerr << "Optioin for derivative is not recognized.\n";
457  }
458 
459 }
