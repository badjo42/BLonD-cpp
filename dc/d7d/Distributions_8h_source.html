20 void longitudinal_bigaussian(ftype sigma_dt, ftype sigma_dE = 0,
21  int seed = 0, bool reinsertion = false)
22 {
23  auto GP = Context::GP;
24  auto RfP = Context::RfP;
25  auto Beam = Context::Beam;
26  if (GP->n_sections > 1) {
27  std::cerr << "WARNING: longitudinal_bigaussian is not yet properly\n"
28  << "computed for several sections!\n";
29  }
30  if (RfP->n_rf > 1) {
31  std::cerr << "WARNING: longitudinal_bigaussian for multiple RF\n"
32  << "is not yet implemented\n";
33  }
34 
35  uint counter = RfP->counter;
36  ftype harmonic = RfP->harmonic[0][counter];
37  ftype energy = GP->energy[0][counter];
38  ftype beta = GP->beta[0][counter];
39  ftype omega_RF = RfP->omega_RF[0][counter];
40  ftype phi_s = RfP->phi_s[counter];
41  ftype phi_RF = RfP->phi_RF[0][counter];
42  ftype eta0 = RfP->eta_0(counter);
43 
44  if (sigma_dE == 0) {
45  auto voltage = GP->charge * RfP->voltage[0][counter];
46  auto phi_b = omega_RF * sigma_dt + phi_s;
47  sigma_dE =
48  sqrt(voltage * energy * beta * beta *
49  (cos(phi_b) - cos(phi_s) + (phi_b - phi_s) * sin(phi_s)) /
50  (constant::pi * harmonic * eta0));
51  }
52 
53  Beam->sigma_dE = sigma_dE;
54  Beam->sigma_dt = sigma_dt;
55 
56  if (seed < 0) {
57  f_vector_t random;
58  util::read_vector_from_file(random, TEST_FILES "/normal_distribution.dat");
59  for (uint i = 0; i < Beam->n_macroparticles; ++i) {
60  ftype r = random[i % random.size()];
61  if (eta0 > 0)
62  Beam->dt[i] = sigma_dt * r
63  + (phi_s - phi_RF) / omega_RF;
64  else
65  Beam->dt[i] = sigma_dt * r
66  + (phi_s - phi_RF - constant::pi) / omega_RF;
67  Beam->dE[i] = sigma_dE * r;
68  }
69  } else {
70  std::default_random_engine generator(seed);
71  std::normal_distribution<ftype> distribution(0.0, 1.0);
72  for (uint i = 0; i < Beam->n_macroparticles; ++i) {
73  ftype r = distribution(generator);
74  if (eta0 > 0)
75  Beam->dt[i] = sigma_dt * r + (phi_s - phi_RF) / omega_RF;
76  else
77  Beam->dt[i] =
78  sigma_dt * r + (phi_s - phi_RF - constant::pi) / omega_RF;
79  r = distribution(generator);
80  Beam->dE[i] = sigma_dE * r;
81  }
82  }
83 
84  // TODO if reinsertion == true
85  if (reinsertion) {
86  ;
87  }
88 }
