20 inline void longitudinal_bigaussian(ftype sigma_dt, ftype sigma_dE = 0,
21  int seed = 0, bool reinsertion = false)
22 {
23  auto GP = Context::GP;
24  auto RfP = Context::RfP;
25  auto Beam = Context::Beam;
26  if (GP->n_sections > 1) {
27  std::cerr << "WARNING: longitudinal_bigaussian is not yet properly\n"
28  << "computed for several sections!\n";
29  }
30  if (RfP->n_rf > 1) {
31  std::cerr << "WARNING: longitudinal_bigaussian for multiple RF\n"
32  << "is not yet implemented\n";
33  }
34 
35  uint counter = RfP->counter;
36  ftype harmonic = RfP->harmonic[0][counter];
37  ftype energy = GP->energy[0][counter];
38  ftype beta = GP->beta[0][counter];
39  ftype omega_RF = RfP->omega_RF[0][counter];
40  ftype phi_s = RfP->phi_s[counter];
41  ftype phi_RF = RfP->phi_RF[0][counter];
42  ftype eta0 = RfP->eta_0(counter);
43 
44  if (sigma_dE == 0) {
45  auto voltage = GP->charge * RfP->voltage[0][counter];
46  auto phi_b = omega_RF * sigma_dt + phi_s;
47  sigma_dE =
48  sqrt(voltage * energy * beta * beta *
49  (cos(phi_b) - cos(phi_s) + (phi_b - phi_s) * sin(phi_s)) /
50  (constant::pi * harmonic * eta0));
51  }
52 
53  Beam->sigma_dE = sigma_dE;
54  Beam->sigma_dt = sigma_dt;
55  // std::cout << "eta0: " << eta0 << "\n";
56  // std::cout << sigma_dE << "\n";
57  // std::cout << sigma_dt << "\n";
58  // std::cout << (phi_s - phi_RF) / omega_RF << "\n";
59  if (seed < 0) {
60  f_vector_t random;
61  // std::string home = util::GETENV("TEST_FILES");
62  util::read_vector_from_file(random, TEST_FILES "/normal_distribution.dat");
63  for (uint i = 0; i < Beam->n_macroparticles; ++i) {
64  ftype r = random[i % random.size()];
65  // ftype r = distribution(generator);
66  if (eta0 > 0)
67  Beam->dt[i] = sigma_dt * r
68  + (phi_s - phi_RF) / omega_RF;
69  else
70  Beam->dt[i] = sigma_dt * r
71  + (phi_s - phi_RF - constant::pi) / omega_RF;
72  Beam->dE[i] = sigma_dE * r;
73  // dprintf("beam_dE: %.8lf \n", Beam->dE[i]);
74  }
75  } else {
76  std::default_random_engine generator(seed);
77  std::normal_distribution<ftype> distribution(0.0, 1.0);
78  for (uint i = 0; i < Beam->n_macroparticles; ++i) {
79  ftype r = distribution(generator);
80  if (eta0 > 0)
81  Beam->dt[i] = sigma_dt * r + (phi_s - phi_RF) / omega_RF;
82  else
83  Beam->dt[i] =
84  sigma_dt * r + (phi_s - phi_RF - constant::pi) / omega_RF;
85  r = distribution(generator);
86  Beam->dE[i] = sigma_dE * r;
87  }
88  // for (uint i = 0; i < Beam->n_macroparticles; ++i)
89  // Beam->dE[i] = sigma_dE * distribution(generator);
90  }
91 
92  // TODO if reinsertion == true
93  if (reinsertion) {
94  ;
95  }
96 }
