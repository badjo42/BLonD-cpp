96 ftype RfParameters::eta_tracking(const Beams *beam, const uint counter,
97  const ftype dE) const
98 {
99  auto GP = Context::GP;
100 
101  ftype eta = 0;
102  if (GP->alpha_order == 1)
103  eta = eta_0(counter);
104  else {
105  ftype delta =
106  dE / ((GP->beta[0][0]) * (GP->beta[0][0]) * GP->energy[0][0]);
107  eta += eta_0(counter) * 1;
108  if (GP->alpha_order > 0)
109  eta += eta_1(counter) * delta;
110  if (GP->alpha_order > 1)
111  eta += eta_2(counter) * delta * delta;
112  if (GP->alpha_order > 2)
113  dprintf(
114  "WARNING: Momentum compaction factor is implemented only up to "
115  "2nd order");
116  }
117  return eta;
118 }

144 void calc_phi_s(ftype *out, RfParameters *rfp,
145  RfParameters::accelerating_systems_t acc_sys)
146 {
147  /*
148  | *The synchronous phase calculated from the rate of momentum change.*
149  | *Below transition, for decelerating bucket: phi_s is in (-Pi/2,0)*
150  | *Below transition, for accelerating bucket: phi_s is in (0,Pi/2)*
151  | *Above transition, for accelerating bucket: phi_s is in (Pi/2,Pi)*
152  | *Above transition, for decelerating bucket: phi_s is in (Pi,3Pi/2)*
153  | *The synchronous phase is calculated at a certain moment.*
154  | *Uses beta, energy averaged over the turn.*
155  */
156  auto GP = Context::GP;
157 
158  auto n_turns = GP->n_turns;
159  auto n_rf = rfp->n_rf;
160  // ftype eta0 = rf_params->eta0;
161  if (acc_sys == RfParameters::accelerating_systems_t::as_single) {
162 
163  ftype *denergy = new ftype[n_turns + 1];
164  for (uint j = 0; j < n_turns; ++j)
165  denergy[j] = rfp->E_increment[j];
166  denergy[n_turns] = rfp->E_increment[n_turns - 1];
167 
168  ftype *acceleration_ratio = new ftype[n_turns + 1];
169  for (uint i = 0; i < n_turns + 1; ++i)
170  acceleration_ratio[i] =
171  denergy[i] / (GP->charge * rfp->voltage[rfp->idx][i]);
172 
173  for (uint i = 0; i < n_turns + 1; ++i)
174  if (acceleration_ratio[i] > 1 || acceleration_ratio[i] < -1)
175  dprintf("Warning!!! Acceleration is not possible (momentum "
176  "increment "
177  "is too big or voltage too low) at index %d\n",
178  i);
179 
180  for (uint i = 0; i < n_turns + 1; ++i)
181  out[i] = asin(acceleration_ratio[i]);
182 
183  // ftype *eta0_middle_points = new ftype[n_turns +1];
184  for (uint i = 0; i < n_turns; ++i) {
185  ftype middle = (rfp->eta_0(i) + rfp->eta_0(i + 1)) / 2;
186  if (middle > 0)
187  out[i] = constant::pi - out[i];
188  else

190  }
191  if (rfp->eta_0(n_turns) > 0)
192  out[n_turns] = constant::pi - out[n_turns];
193  else

199  return;
